Tastier Grammar

Tastier =    "program" ident "{" {VarDecl | ConstDef | ArrayDecl } {ProcDecl} "}"


VarDecl =    Type ident {"," ident} ";"

ConstDecl =  "constant" Type ident "<-" Expr ";"

ArrayDecl =  "array" Type ident "[" number "]" ";"

ProcDecl =   "void" ident "(" ")"

             "{" {VarDecl | ArrayDecl} {ProcDecl} Stat {Stat} "}"


Type =       "int" | "bool"


Stat =     ident ( "[" number "]" ":=" Expr ";" | ":=" Expr [ "?" Expr1 ":" Expr2 ] | "(" ")" ";")

           | ident ":="[ Expr "?" Expr1 ":" Expr2 ]";"

           | "for" "(" Stat; Stat; Stat; ")" "do" "{" Stat "}" ";"

           | "switch" "(" Expr ")" "{" "case" "(" Expr ")" ":" [ Stat; "break" ";" ] { "{" "case" "(" Expr ")" ":" [ Stat; "break" ";"]"}" } "default" ":" [Stat; break;] "}"

           | "if" Expr Stat ["else" Stat]

           | "while Expr Stat

           | "read" ident ";"

           | "write" Expr | string ";"

           | "writeln" Expr ";"

           | "{" {VarDecl | ArrayDecl} Stat {Stat} ”}”


Expr =       SimExpr [RelOp SimExpr]

SimExpr =    Term {AddOp Term}

Term =       Primary {MulOp Primary}

Primary =    (ident | ident "[" number "]" | number | "-" Primary 

             | "true" | "false" | "(" Expr ")")


AddOp =      "+" | "-"

MulOp =      "*" | "div" | "DIV" | "mod" | "MOD"

RelOp =      "=" | "<" | ">" | "|=" | "<=" | ">="
